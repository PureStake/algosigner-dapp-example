Sign Asset Config Transaction See code on GitHub

Ask the user to sign an asset creation transaction using AlgoSigner's Sign method 

response
AlgoSigner.sign({
  from: document.getElementById('from').value,
  assetName: document.getElementById('name').value,
  assetUnitName: document.getElementById('unit-name').value,
  assetTotal: +document.getElementById('total').value,
  assetDecimals: +document.getElementById('decimals').value,
  note: document.getElementById('note').value,
  type: 'acfg',
  fee: txParams['min-fee'],
  firstRound: txParams['last-round'],
  lastRound: txParams['last-round'] + 1000,
  genesisID: txParams['genesis-id'],
  genesisHash: txParams['genesis-hash']
})
.then((d) => {
  signedTx = d;
})
.catch((e) => {
  console.error(e);
});
